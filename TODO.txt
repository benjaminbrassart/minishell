memory leaks in redirections without command
	solution: see next

unhandled:
	echo 123 | > test.log
	echo 123 | < test.log
	echo 123 | << EOF
	echo 123 | >> test.log

	expected:	no output and file open then closed
	current:	segfault

	solution:	handle commands differently ->
		a command is either word(s), redirection(s) or both
